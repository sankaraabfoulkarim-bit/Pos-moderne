⚡ Supabase

Non vérifié

Supabase stocke toutes les données marchands (PostgreSQL). Les clés sont configurées côté serveur (variables Vercel) — aucune clé n'est stockée dans cette interface. Gérez votre projet sur supabase.com/dashboard.

📋 Schéma SQL Supabase

Copiez ce schéma dans Supabase Dashboard → SQL Editor pour créer les tables : 

create table if not exists merchants (
  id text primary key, nom_commerce text not null,
  proprietaire text, telephone text unique,
  password text, licence text default 'active',
  licence_expiry timestamptz, actif boolean default true,
  plan_type text default 'mensuel', created_at timestamptz default now()
);
create table if not exists configs (
  id text primary key, merchant_id text references merchants(id),
  couleur_theme text, devise text default 'FCFA',
  pin text default '', created_at timestamptz default now()
);
create table if not exists products (
  id text primary key, merchant_id text references merchants(id),
  nom text, prix numeric, stock integer, categorie text,
  created_at timestamptz default now()
);
create table if not exists clients (
  id text primary key, merchant_id text references merchants(id),
  nom text, telephone text, whatsapp text,
  created_at timestamptz default now()
);
create table if not exists sales (
  id text primary key, merchant_id text references merchants(id),
  client_nom text, items jsonb, total numeric,
  created_at timestamptz default now()
);
create table if not exists payments (
  id text primary key, phone text, amount numeric,
  transaction_id text unique, operator text,
  merchant_id text, plan text, days integer,
  statut text, created_at timestamptz default now()
);
-- Activer RLS sur toutes les tables
alter table merchants enable row level security;
alter table configs   enable row level security;
alter table products  enable row level security;
alter table clients   enable row level security;
alter table sales     enable row level security;
alter table payments  enable row level security;
